Q1 Top 5 Engineering salaries
Retrieve the first name, last name, and salary of employees in the Engineering department, ordered by salary descending. Return only top 5.
SELECT first_name, last_name, salary
FROM   employees
WHERE  department = 'Engineering'
ORDER  BY salary DESC
LIMIT  5;

Q1 Revenue per category (completed)
Find total revenue per category for completed orders only, sorted by total revenue descending.
SELECT category,
       SUM(quantity * unit_price) AS total_revenue
FROM   orders
WHERE  status = 'completed'
GROUP  BY category
ORDER  BY total_revenue DESC;
Q2 High-demand products (qty > 5)
Find all products where total quantity ordered > 5. Show product and total_quantity, sorted descending.
SELECT product,
       SUM(quantity) AS total_quantity
FROM   orders
GROUP  BY product
HAVING SUM(quantity) > 5
ORDER  BY total_quantity DESC;
Q3 Monthly sales summary
Find total revenue and order count per month for completed orders only. Sort by month ascending.
SELECT EXTRACT(MONTH FROM order_date) AS month,
       COUNT(*)                        AS total_orders,
       SUM(unit_price * quantity)      AS total_revenue
FROM   orders
WHERE  status = 'completed'
GROUP  BY EXTRACT(MONTH FROM order_date)
ORDER  BY month ASC;
Q4 Avg order value > 30k
Find categories where average order value > 30,000. Show category and avg_order_value, sorted descending.
SELECT category,
       AVG(quantity * unit_price) AS avg_order_value
FROM   orders
GROUP  BY category
HAVING AVG(quantity * unit_price) > 30000
ORDER  BY avg_order_value;

Q1 INNER JOIN — emp + departments
Retrieve employee name, salary and department name for all employees that belong to a department. Sort by department name ascending.
SELECT e.name,
       e.salary,
       d.dept_name
FROM   emp e
       INNER JOIN departments d ON e.dept_id = d.dept_id
ORDER  BY d.dept_name ASC;
Q2 LEFT JOIN — include unassigned employees
Retrieve all employees names and their department name. If an employee has no department, show NULL. Sort by employee name ascending.
SELECT e.name,
       d.dept_name
FROM   emp e
       LEFT JOIN departments d ON e.dept_id = d.dept_id
ORDER  BY e.name ASC;
Q3 SELF JOIN — employee + manager
Retrieve each employee's name and their manager's name. Employees with no manager should still appear with NULL. Sort by employee name ascending.
SELECT e.name,
       e1.name AS manager_name
FROM   emp e
       LEFT JOIN emp e1 ON e.manager_id = e1.emp_id
ORDER  BY e.name;

Q4 Dept headcount + avg salary
Retrieve dept_name, total_employees, avg_salary for all departments including 0-employee ones. Round avg_salary to 2 decimal places.
SELECT d.dept_name,
       SUM(CASE
               WHEN e.emp_id IS NULL THEN 0
               ELSE 1
           END)               AS total_employees,
       ROUND(AVG(e.salary), 2) AS avg_salary
FROM   departments d
       LEFT JOIN emp e ON d.dept_id = e.dept_id
GROUP  BY d.dept_id
ORDER  BY total_employees ASC;

Q1 RANK() salary within department
For each employee show name, dept_name, salary and RANK() within their department by salary descending. Sort by dept_name asc, rank asc.
SELECT e.name,
       d.dept_name,
       e.salary,
       DENSE_RANK() OVER (PARTITION BY d.dept_id ORDER BY e.salary DESC) AS rank_number
FROM   emp e
       LEFT JOIN departments d ON e.dept_id = d.dept_id
ORDER  BY d.dept_name ASC,
          rank_number    ASC;
Q2 DENSE_RANK() — no gaps
Same as above but use DENSE_RANK() — no gaps in ranking for tied salaries.
SELECT e.name,
       d.dept_name,
       e.salary,
       DENSE_RANK() OVER (PARTITION BY d.dept_id ORDER BY e.salary DESC) AS rank_number
FROM   emp e
       LEFT JOIN departments d ON e.dept_id = d.dept_id
ORDER  BY d.dept_name ASC,
          rank_number ASC;
Q3 LAG() — previous month revenue
Show each month, its total revenue, and the previous month's revenue using LAG(). Sort by month ascending.
WITH monthly_revenue AS (
    SELECT EXTRACT(MONTH FROM s.sale_date) AS month,
           SUM(s.quantity * p.price)        AS total_revenue
    FROM   sales s
           LEFT JOIN products p ON s.product_id = p.product_id
    GROUP  BY month
)
SELECT month,
       total_revenue,
       LAG(total_revenue, 1) OVER (ORDER BY month ASC) AS prev_month_revenue
FROM   monthly_revenue;
